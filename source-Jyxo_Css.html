 35:     /**
 36:      * Cleans up a CSS stylesheet.
 37:      *
 38:      * @param string $css Stylesheet definition
 39:      * @return string
 40:      */
 41:     public static function repair($css)
 42:     {
 43:         // Convert properties to lowercase
 44:         $css = preg_replace_callback('~((?:^|\{|;)\s*)([-a-z]+)(\s*:)~i', function($matches) {
 45:             return $matches[1] . strtolower($matches[2]) . $matches[3];
 46:         }, $css);
 47:         // Convert rgb() and url() to lowercase
 48:         $css = preg_replace_callback('~(rgb|url)(?=\s*\()~i', function($matches) {
 49:             return strtolower($matches[1]);
 50:         }, $css);
 51:         // Remove properties without values
 52:         $css = preg_replace_callback('~\s*[-a-z]+\s*:\s*([;}]|$)~i', function($matches) {
 53:             return '}' === $matches[1] ? '}' : '';
 54:         }, $css);
 55:         // Remove MS Office properties
 56:         $css = preg_replace('~\s*mso-[-a-z]+\s*:[^;}]*;?~i', '', $css);
 57:         // Convert color definitions to lowercase
 58:         $css = preg_replace_callback('~(:[^:]*?)(#[abcdef0-9]{3,6})~i', function($matches) {
 59:             return $matches[1] . strtolower($matches[2]);
 60:         }, $css);
 61:         // Convert colors from RGB to HEX
 62:         $css = preg_replace_callback('~rgb\s*\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*\)~', function($matches) {
 63:             return sprintf('#%02x%02x%02x', $matches[1], $matches[2], $matches[3]);
 64:         }, $css);
 65: 
 66:         return $css;
 67:     }

 69:     /**
 70:      * Filters given properties.
 71:      *
 72:      * @param string $css Stylesheet definition
 73:      * @param array $properties Filtered properties
 74:      * @param boolean $exclude If true, $properties will be removed from the stylesheet; if false, only $properties will be left
 75:      * @return string
 76:      */
 77:     public static function filterProperties($css, array $properties, $exclude = true)
 78:     {
 79:         $properties = array_flip($properties);
 80:         return preg_replace_callback('~\s*([-a-z]+)\s*:[^;}]*;?~i', function($matches) use ($properties, $exclude) {
 81:             if ($exclude) {
 82:                 return isset($properties[$matches[1]]) ? '' : $matches[0];
 83:             } else {
 84:                 return isset($properties[$matches[1]]) ? $matches[0] : '';
 85:             }
 86:         }, $css);
 87:     }

 89:     /**
 90:      * Removes unnecessary characters from a CSS stylesheet.
 91:      *
 92:      * It is recommended to use the repair() method on the stylesheet definition first.
 93:      *
 94:      * @param string $css Stylesheet definition
 95:      * @return string
 96:      */
 97:     public static function pack($css)
 98:     {
 99:         // Comments
100:         $packed = preg_replace('~/\*.*\*/~sU', '', $css);
101:         // Whitespace
102:         $packed = preg_replace('~\s*([>+\~,{:;}])\s*~', '\1', $packed);
103:         $packed = preg_replace('~\(\s+~', '(', $packed);
104:         $packed = preg_replace('~\s+\)~', ')', $packed);
105:         $packed = trim($packed);
106:         // Convert colors from #ffffff to #fff
107:         $packed = preg_replace('~(:[^:]*?#)([abcdef0-9]{1})\2([abcdef0-9]{1})\3([abcdef0-9]{1})\4~', '\1\2\3\4', $packed);
108:         // Empty selectors
109:         $packed = preg_replace('~(?<=})[^{]+\{\}~', '', $packed);
110:         // Remove units when 0
111:         $packed = preg_replace('~([\s:]0)(?:px|pt|pc|in|mm|cm|em|ex|%)~', '\1', $packed);
112:         // Unnecessary semicolons
113:         $packed = str_replace(';}', '}', $packed);
114:         $packed = trim($packed, ';');
115: 
116:         return $packed;
117:     }

119:     /**
120:      * Converts HTML styles inside <style> elements to inline styles.
121:      * Supported selectors:
122:      *  a {...}
123:      *  #header {...}
124:      *  .icon {...}
125:      *  h1#header {...}
126:      *  a.icon.small {...}
127:      *  a#remove.icon.small {...}
128:      *  a img {...}
129:      *  h1, h2 {...}
130:      *  a:link {...} - converts to a {...}
131:      *
132:      * @param string $html Processed HTML source
133:      * @return string
134:      */
135:     public static function convertStyleToInline($html)
136:     {
137:         // Extract styles from the source
138:         $cssList = self::parseStyle($html);
139: 
140:         // If no styles were found, return the original HTML source
141:         if (empty($cssList)) {
142:             return $html;
143:         }
144: 
145:         // Parse the HTML source
146:         preg_match_all('~(?:<\w+[^>]*(?: /)?>)|(?:</\w+>)|(?:<![^>]+>)|(?:[^<]+)~', $html, $matches);
147:         $path = array();
148:         $html = '';
149:         $inStyle = false;
150:         foreach ($matches[0] as $htmlPart) {
151:             // Skip <style> elements
152:             if (0 === strpos($htmlPart, '<style')) {
153:                 // <style> opening tag
154:                 $inStyle = true;
155:             } elseif (0 === strpos($htmlPart, '</style')) {
156:                 // <style> closing tag
157:                 $inStyle = false;
158:             } elseif (!$inStyle) {
159:                 // Not inside the <style> element
160: 
161:                 // Closing tag
162:                 if (0 === strpos($htmlPart, '</')) {
163:                     array_pop($path);
164:                 } elseif (('<' === $htmlPart[0]) && (0 !== strpos($htmlPart, '<!'))) {
165:                     // Opening tag or empty element, ignoring comments
166: 
167:                     $htmlPart = trim($htmlPart, '/<> ');
168: 
169:                     $attributeList = array('id' => '', 'class' => '');
170:                     // If there is no space, there are no attributes
171:                     if (false !== strpos($htmlPart, ' ')) {
172:                         list($tag, $attributes) = explode(' ', $htmlPart, 2);
173: 
174:                         // Parse attributes
175:                         foreach (explode('" ', $attributes) as $attribute) {
176:                             list($attributeName, $attributeContent) = explode('="', $attribute);
177:                             $attributeList[$attributeName] = trim($attributeContent, '"');
178:                         }
179:                     } else {
180:                         $tag = $htmlPart;
181:                     }
182:                     $attributeClass = !empty($attributeList['class']) ? explode(' ', $attributeList['class']) : array();
183: 
184:                     // Add element information to the path
185:                     array_push($path, array(
186:                             'tag' => $tag,
187:                             'id' => $attributeList['id'],
188:                             'class' => $attributeClass
189:                         )
190:                     );
191: 
192:                     // Walk through the CSS definition list and add applicable properties
193:                     // Because of inheritance, walk the array in reversed order
194:                     foreach (array_reverse($cssList) as $css) {
195:                         // Selectors have to have equal or less parts than the HTML element nesting level
196:                         if (count($css['selector']) > count($path)) {
197:                             continue;
198:                         }
199: 
200:                         // The last selector part must correspond to the last processed tag
201:                         $lastSelector = end($css['selector']);
202:                         if (((!empty($lastSelector['tag'])) && ($tag !== $lastSelector['tag']))
203:                                 || ((!empty($lastSelector['id'])) && ($attributeList['id'] !== $lastSelector['id']))
204:                                 || (count(array_diff($lastSelector['class'], $attributeClass)) > 0)) {
205:                             continue;
206:                         }
207: 
208:                         $add = true;
209:                         $lastPathKey = count($path);
210:                         foreach (array_reverse($css['selector']) as $selector) {
211:                             // Nothing was found in the previous search, no reason to continue searching
212:                             if (!$add) {
213:                                 break;
214:                             }
215: 
216:                             for ($i = ($lastPathKey - 1); $i >= 0; $i--) {
217:                                 if (((empty($selector['tag'])) || ($path[$i]['tag'] === $selector['tag']))
218:                                         && ((empty($selector['id'])) || ($path[$i]['id'] === $selector['id']))
219:                                         && (0 === count(array_diff($selector['class'], $path[$i]['class'])))) {
220:                                     $lastPathKey = $i;
221:                                     continue 2;
222:                                 }
223:                             }
224: 
225:                             $add = false;
226:                         }
227: 
228:                         if ($add) {
229:                             // Add a semicolon if missing
230:                             if (';' !== substr($css['rules'], -1)) {
231:                                 $css['rules'] .= ';';
232:                             }
233:                             // CSS is processed in the reversed order, co place to the beginning
234:                             $attributeList['style'] = $css['rules'] . (isset($attributeList['style']) ? $attributeList['style'] : '');
235:                         }
236:                     }
237: 
238:                     // Creates a new tag
239:                     $htmlPart = '<' . $tag;
240:                     foreach ($attributeList as $attributeName => $attributeContent) {
241:                         // Not using empty() because it would make attributes with the "0" value ignored
242:                         if ('' !== $attributeContent) {
243:                             $htmlPart .= ' ' . $attributeName . '="' . $attributeContent . '"';
244:                         }
245:                     }
246: 
247:                     // Empty tags
248:                     switch ($tag) {
249:                         case 'br':
250:                         case 'hr':
251:                         case 'img':
252:                         case 'input':
253:                             $htmlPart .= ' />';
254:                             break;
255:                         default:
256:                             $htmlPart .= '>';
257:                             break;
258:                     }
259:                 }
260:             }
261: 
262:             // Append the part to the HTML source
263:             $html .= $htmlPart;
264:         }
265: 
266:         // In case of float: add a cleaner (if there is none present already)
267:         $cleaner = '<div style="clear: both; visibility: hidden; overflow: hidden; height: 1px;">';
268:         if ((preg_match('~<\w+[^>]+style="[^"]*float:[^"]*"~', $html))
269:                 && (!preg_match('~' . preg_quote($cleaner) . '\s*</body>\s*</html>\s*$~', $html))) {
270:             $html = str_replace('</body>', $cleaner . '</body>', $html);
271:         }
272: 
273:         return $html;
274:     }

276:     /**
277:      * Helper method for searching and parsing <style> definitions inside a HTML source.
278:      *
279:      * @param string $html Processed HTML source
280:      * @return array
281:      * @see Jyxo_Css::convertStyleToInline()
282:      */
283:     private static function parseStyle($html)
284:     {
285:         // Find <style> elements
286:         if (!preg_match_all('~<style\s+(?:[^>]+\s+)*type="text/css"[^>]*>(.*?)</style>~s', $html, $styles)) {
287:             return array();
288:         }
289: 
290:         $cssList = array();
291:         foreach ($styles[1] as $style) {
292:             // Remove CDATA and HTML comments
293:             $style = str_replace(array('<![CDATA[', ']]>', '<!--', '-->'), '', $style);
294: 
295:             // Optimize the parsed definitions
296:             $style = self::pack($style);
297: 
298:             if (empty($style)) {
299:                 continue;
300:             }
301: 
302:             // Replace quotes with apostrophes
303:             $style = str_replace('"', "'", $style);
304: 
305:             // Remove the last empty part
306:             $definitions = explode('}', $style, -1);
307: 
308:             foreach ($definitions as $definition) {
309:                 // Allows only supported selectors with valid rules
310:                 if (!preg_match('~^(?:(?:(?:[-_\w#.:]+)\s?)+,?)+{(?:[-\w]+:[^;]+[;]?)+$~', $definition)) {
311:                     continue;
312:                 }
313: 
314:                 list($selector, $rules) = explode('{', $definition);
315:                 foreach (explode(',', $selector) as $part) {
316:                     // Convert a:link to a
317:                     $part = str_replace(':link', '', $part);
318: 
319:                     $parsedSelector = array();
320:                     foreach (explode(' ', $part) as $selectorPart) {
321:                         // If no tag name was given use a fake one
322:                         if (('.' === $selectorPart[0]) || ('#' === $selectorPart[0])) {
323:                             $selectorPart = ' ' . $selectorPart;
324:                         }
325: 
326:                         if (false !== strpos($selectorPart, '.')) {
327:                             list($selectorPart, $class) = explode('.', $selectorPart, 2);
328:                             // There can be multiple classes
329:                             $class = explode('.', $class);
330:                         } else {
331:                             $class = array();
332:                         }
333:                         if (false !== strpos($selectorPart, '#')) {
334:                             list($selectorPart, $id) = explode('#', $selectorPart, 2);
335:                         } else {
336:                             $id = '';
337:                         }
338:                         $tag = trim($selectorPart);
339: 
340:                         $parsedSelector[] = array(
341:                             'tag' => strtolower($tag),
342:                             'id' => $id,
343:                             'class' => $class
344:                         );
345:                     }
346: 
347:                     $cssList[] = array(
348:                         'selector' => $parsedSelector,
349:                         'rules' => $rules
350:                     );
351:                 }
352:             }
353:         }
354: 
355:         return $cssList;
356:     }
